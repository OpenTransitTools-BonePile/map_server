geoserver
==========

install:
  1. install java 8
  1. grab geoserver -- https://docs.geoserver.org/latest/en/user/installation/index.html#installation
  1. grab css plugin -- https://docs.geoserver.org/latest/en/user/styling/css/install.html
  1. export DATA_DIR = ...
  1. cd geoserver/bin
  1. sh startup.sh 

run:
  1. scripts/run.sh
